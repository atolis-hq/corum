# Web App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a browser-based web UI served by the MCP server process, driven entirely by graph templates and node data with no hardcoded layouts.

**Architecture:** An Express HTTP server runs in the same Node.js process as the MCP stdio transport, sharing the loaded graph. The frontend is CDN React + Babel (no build step), fetching data from a REST API at `/api/*` that calls the same graph functions as the MCP tools. A `--no-web` CLI flag disables the HTTP server; a separate `npm run web` entry point starts it standalone.

**Tech Stack:** Node.js 20+, TypeScript (server only), Express 4, React 18 (CDN), Babel Standalone (CDN), Node built-in test runner.

- [ ] **Step 2: Create `src/web/server.ts` with health endpoint**

```typescript
import express from 'express'
import path from 'node:path'
import { fileURLToPath, pathToFileURL } from 'node:url'
import { readdir } from 'node:fs/promises'
import type { AddressInfo } from 'node:net'
import { getCluster, listNodes, type ListNodesFilter } from '../graph/index.js'
import { loadGraph } from '../loader/index.js'
import type { Graph } from '../schema/index.js'
import { QueryError } from '../schema/index.js'

const __filename = fileURLToPath(import.meta.url)
const __dirname = path.dirname(__filename)
const WEB_DIR = path.join(__dirname, '..', '..', '..', 'web')

export type WebServerOptions = {
  port?: number
}

export type WebServerHandle = {
  port: number
  close(): Promise<void>
}

export function createApp(graph: Graph): express.Application {
  const app = express()

  app.get('/health', (_req, res) => {
    res.json({ ok: true })
  })

  app.use(express.static(WEB_DIR, {
    setHeaders(res, filePath) {
      if (filePath.endsWith('.jsx')) {
        res.setHeader('Content-Type', 'text/babel; charset=utf-8')
      }
    },
  }))

  return app
}

export function startWebServer(graph: Graph, options: WebServerOptions = {}): Promise<WebServerHandle> {
  const port = options.port ?? parseInt(process.env.CORUM_WEB_PORT ?? '3000', 10)
  const app = createApp(graph)
  return new Promise((resolve, reject) => {
    const server = app.listen(port, () => {
      const addr = server.address() as AddressInfo
      if (options.port !== 0) {
        console.error(`[corum web] http://localhost:${addr.port}`)
      }
      resolve({
        port: addr.port,
        close: () => new Promise<void>((res, rej) => server.close(err => (err ? rej(err) : res()))),
      })
    })
    server.on('error', reject)
  })
}

function isEntrypoint(): boolean {
  return process.argv[1] !== undefined && import.meta.url === pathToFileURL(process.argv[1]).href
}

if (isEntrypoint()) {
  const graphPath = process.env.CORUM_GRAPH_PATH ?? path.join(process.cwd(), '.corum/graph')
  const graph = await loadGraph({ graphPath, strict: true })
  await startWebServer(graph)
}
```

function parseRoute() {
  const hash = window.location.hash.slice(1) || '/dashboard';
  const [pathname, search] = hash.split('?');
  return { pathname, params: new URLSearchParams(search) };
}

function navigate(path) {
  window.location.hash = path;
}

/* ── Nav helpers ───────────────────────────────────────────────── */

function displayName(id) {
  const parts = id.split('.');
  return parts[parts.length - 1];
}

function buildNavTree(nodes) {
  // Map<component, Map<templateName, node[]>>
  const tree = new Map();
  for (const node of nodes) {
    if (!tree.has(node.component)) tree.set(node.component, new Map());
    const comp = tree.get(node.component);
    if (!comp.has(node.template)) comp.set(node.template, []);
    comp.get(node.template).push(node);
  }
  return tree;
}

/* ── Top bar ───────────────────────────────────────────────────── */

function NavRail({ activeSection, onSection }) {
  const items = [
    { id: 'dashboard', icon: 'grid', label: 'Dashboard' },
    { id: 'components', icon: 'cube', label: 'Components' },
  ];
  return (
    <div className="nav-rail">
      {items.map(item => (
        <div
          key={item.id}
          className={`nav-rail-item${activeSection === item.id ? ' active' : ''}`}
          onClick={() => onSection(item.id)}
        >
          <Icon name={item.icon} size={16} />
          <span>{item.label}</span>
        </div>
      ))}
    </div>
  );
}

function NavTree({ navTree, templates, activeNodeId, onNode }) {
  const [openComponents, setOpenComponents] = useState(() => {
    const initial = {};
    for (const comp of navTree.keys()) initial[comp] = true;
    return initial;
  });

  function toggleComponent(comp) {
    setOpenComponents(prev => ({ ...prev, [comp]: !prev[comp] }));
  }

  const templateMap = new Map(templates.map(t => [t.name, t]));

  return (
    <div className="nav-tree">
      {[...navTree.entries()].sort(([a], [b]) => a.localeCompare(b)).map(([comp, templateGroups]) => (
        <div key={comp}>
          <div className="nav-section-head" onClick={() => toggleComponent(comp)}>
            <span>{comp}</span>
            <Icon name={openComponents[comp] ? 'caret-down' : 'caret'} size={12} />
          </div>
          {openComponents[comp] && [...templateGroups.entries()].sort(([a], [b]) => a.localeCompare(b)).map(([tmplName, nodes]) => {
            const tmpl = templateMap.get(tmplName);
            const colour = tmpl?.ui?.colour ?? 'var(--ink-4)';
            return (
              <div key={tmplName}>
                <div className="nav-template-head">
                  <div className="nav-template-accent" style={{ background: colour }} />
                  <span>{tmplName}</span>
                </div>
                {nodes.map(node => (
                  <div
                    key={node.id}
                    className={`nav-node-item${node.id === activeNodeId ? ' active' : ''}`}
                    onClick={() => onNode(node.id)}
                    title={node.id}
                  >
                    {displayName(node.id)}
                  </div>
                ))}
              </div>
            );
          })}
        </div>
      ))}
    </div>
  );
}

function App() {
  const [templates, setTemplates] = useState([]);
  const [nodes, setNodes] = useState([]);
  const [loading, setLoading] = useState(true);
  const [route, setRoute] = useState(parseRoute);

  useEffect(() => {
    Promise.all([
      fetch('/api/templates').then(r => r.json()),
      fetch('/api/nodes').then(r => r.json()),
    ]).then(([tmplData, nodeData]) => {
      setTemplates(tmplData);
      setNodes(nodeData);
      setLoading(false);
    });
  }, []);

  useEffect(() => {
    const handler = () => setRoute(parseRoute());
    window.addEventListener('hashchange', handler);
    return () => window.removeEventListener('hashchange', handler);
  }, []);

  const activeNodeId = route.pathname === '/node' ? route.params.get('id') : null;
  const activeSection = activeNodeId ? 'components' : (route.pathname.slice(1) || 'dashboard');

  function handleSection(section) {
    navigate(`/${section}`);
  }

  function handleNode(nodeId) {
    navigate(`/node?id=${encodeURIComponent(nodeId)}`);
  }

  const navTree = buildNavTree(nodes);
  const showTree = activeSection === 'components' || activeNodeId;

  return (
    <>
      <TopBar />
      <div className="main">
        <NavRail activeSection={activeSection} onSection={handleSection} />
        {showTree && !loading && (
          <NavTree
            navTree={navTree}
            templates={templates}
            activeNodeId={activeNodeId}
            onNode={handleNode}
          />
        )}
        {loading ? (
          <div className="content"><p className="label-sm">Loading graph…</p></div>

          <DashboardPage />
        ) : route.pathname === '/components' ? (
          <ComponentsPage />
        ) : route.pathname === '/node' ? (
          <NodePage nodeId={activeNodeId} templates={templates} />
        ) : (
          <div className="content"><p className="label-sm">Page not found.</p></div>
        )}
      </div>
    </>
  );
}

function NodePage({ nodeId, templates }) {
  const [cluster, setCluster] = useState(null);
  const [error, setError] = useState(null);

  useEffect(() => {
    if (!nodeId) return;
    setCluster(null);
    setError(null);
    fetch(`/api/cluster?nodeId=${encodeURIComponent(nodeId)}`)
      .then(r => r.ok ? r.json() : Promise.reject(r.status))
      .then(setCluster)
      .catch(e => setError(String(e)));
  }, [nodeId]);

  if (error) return <div className="content"><p style={{ color: 'var(--warn)' }}>Error loading node: {error}</p></div>;
  if (!cluster) return <div className="content"><p className="label-sm">Loading…</p></div>;

  const { root, children } = cluster;
  const template = templates.find(t => t.name === root.template);
  const colour = template?.ui?.colour ?? null;

  // Plugin override check
  const Plugin = window.CorumPlugins?.[root.template];
  if (Plugin) return <Plugin node={root} cluster={cluster} template={template} />;

  // Group all children by template
  const displayChildren = new Map();
  for (const child of children) {
    if (!displayChildren.has(child.template)) displayChildren.set(child.template, []);
    displayChildren.get(child.template).push(child);
  }

  return (
    <div className="content">
      {/* Header */}
      <div style={{ marginBottom: 18 }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: 10, marginBottom: 6 }}>
          <h1 style={{ margin: 0 }}>{displayName(root.id)}</h1>
          <TemplateBadge name={root.template} colour={colour} />
          <StateTag state={root.state} />
          <StabilityTag stability={root.stability} />
        </div>
        <div className="label-sm mono">{root.id}</div>
      </div>

      {/* Meta strip */}
      <div className="meta-strip">
        {[
          ['Component', root.component],
          ['State', root.state],
          ['Stability', root.stability],
          ['Schema version', root.schemaVersion],
          ['Last modified', root.lastModifiedAt],
        ].map(([label, value], i) => (
          <div key={i} className="meta-cell">
            <div className="label-xs">{label}</div>
            <div style={{ fontSize: 12, marginTop: 3 }}>{value}</div>
          </div>
        ))}
      </div>

      {/* Properties card */}
      {Object.keys(root.properties ?? {}).length > 0 && (
        <div className="card">
          <div className="card-head">Properties</div>
          <div className="card-body">
            <PropertiesTable properties={root.properties} />
          </div>
        </div>
      )}

      {/* Schema cards — one per non-core child template group */}
      {[...displayChildren.entries()].map(([tmplName, groupNodes]) => (
        <SchemaCard key={tmplName} title={tmplName} nodes={groupNodes} />
      ))}
    </div>
  );
}
