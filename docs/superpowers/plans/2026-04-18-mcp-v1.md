# MCP v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Load a Corum graph from the filesystem (template packs + cluster files + edge files) into an in-memory graph and expose it via three MCP tools: `list_nodes`, `get_cluster`, `get_linked_fields`.

**Architecture:** Single TypeScript package; three loader phases (packs → clusters → edges); in-memory graph with two indexed Maps; three graph query functions wired to MCP tool handlers.

**Tech Stack:** Node.js v20, TypeScript, `yaml` (already installed), `@modelcontextprotocol/sdk`, Node's built-in `node:test` runner.

The pack loader reads template YAML files from each pack's `templates/` directory, validates required fields (`name`, `version`), applies `_base` owned sections (schemas/enums) to all non-base templates, and resolves `extends` inheritance.

**Structural edge type mapping** used later in cluster loading, defined here as a constant:
```
Field      → has-field  (parent schema emits this edge)
EnumValue  → has-value  (parent enum emits this edge)
```

```typescript
import { readFileSync, readdirSync, existsSync } from 'node:fs'
import path from 'node:path'
import { parse as parseYaml } from 'yaml'
import type { Template, Diagnostic } from '../schema/index.js'

const RESERVED_TEMPLATE_KEYS = new Set([
  'name', 'version', 'core', 'abstract', 'extends',
  'description', 'properties', 'edge-types', 'ui',
])

export function getOwnedSections(template: Template): Record<string, string> {
  const result: Record<string, string> = {}
  for (const [key, value] of Object.entries(template)) {
    if (
      !RESERVED_TEMPLATE_KEYS.has(key) &&
      typeof value === 'object' &&
      value !== null &&
      'item-template' in (value as object)
    ) {
      result[key] = (value as Record<string, string>)['item-template']
    }
  }
  return result
}

export async function loadPacks(
  packDirs: string[],
  diagnostics: Diagnostic[],
): Promise<Map<string, Template>> {
  const templates = new Map<string, Template>()
  let base: Template | undefined

  for (const packDir of packDirs) {
    const templatesDir = path.join(packDir, 'templates')
    if (!existsSync(templatesDir)) {
      diagnostics.push({
        severity: 'warning',
        file: templatesDir,
        message: `templates directory not found in pack: ${packDir}`,
      })
      continue
    }

    const files = readdirSync(templatesDir).filter(f => f.endsWith('.yaml'))

    for (const file of files) {
      const filePath = path.join(templatesDir, file)
      let raw: unknown
      try {
        raw = parseYaml(readFileSync(filePath, 'utf-8'))
      } catch (err) {
        diagnostics.push({ severity: 'error', file: filePath, message: `failed to parse YAML: ${err}` })
        continue
      }

      const t = raw as Record<string, unknown>
      if (typeof t.name !== 'string' || typeof t.version !== 'string') {
        diagnostics.push({ severity: 'error', file: filePath, message: 'template missing required name or version' })
        continue
      }

      const template = t as Template
      if (template.name === 'base') {
        base = template
      } else {
        templates.set(template.name, template)
      }
    }
  }

  // Apply _base owned sections to all templates that don't already declare them
  if (base) {
    const baseSections = getOwnedSections(base)
    for (const template of templates.values()) {
      for (const [key, value] of Object.entries(base)) {
        if (!RESERVED_TEMPLATE_KEYS.has(key) && !(key in template)) {
          ;(template as Record<string, unknown>)[key] = value
        }
      }
    }
  }

  // Resolve extends inheritance
  for (const template of templates.values()) {
    if (template.extends) {
      const parent = templates.get(template.extends)
      if (!parent) {
        diagnostics.push({
          severity: 'error',
          file: `template:${template.name}`,
          message: `extends references unknown template: ${template.extends}`,
        })
        continue
      }
      if (parent.properties && !template.properties) {
        template.properties = parent.properties
      } else if (parent.properties && template.properties) {
        template.properties = { allOf: [parent.properties, template.properties] }
      }
      // Inherit owned sections from parent that child does not declare
      for (const [key, value] of Object.entries(parent)) {
        if (!RESERVED_TEMPLATE_KEYS.has(key) && !(key in template)) {
          ;(template as Record<string, unknown>)[key] = value
        }
      }
    }
  }

  return templates
}
```

The cluster loader walks `{graphPath}/components/{component}/{TemplateFolders}/*.yaml`. For each cluster file it:
1. Creates the root node
2. Recursively materialises owned child nodes (Schema, Field, EnumDefinition, EnumValue, Invariant, DomainOperation, etc.)
3. Materialises `has-field` edges (schema → field) and `has-value` edges (enum → enumValue)

**Expected node count from sample-graph fixtures:**
- `orders.DomainModel.order` cluster: 21 nodes (1 root + 2 schemas + 9 fields + 1 enum + 3 values + 2 invariants + 3 operations)
- `orders.APIEndpoint.create-order` cluster: 24 nodes (1 root + 4 schemas + 15 fields + 1 enum + 3 values)
- **Total: 45 nodes**

**Expected structural edge count:**
- DomainModel cluster: 9 has-field + 3 has-value = 12
- APIEndpoint cluster: 15 has-field + 3 has-value = 18

```typescript
import { readFileSync, readdirSync, existsSync } from 'node:fs'
import path from 'node:path'
import { parse as parseYaml } from 'yaml'
import type { Node, Edge, EdgeType, State, Stability, Template, Diagnostic } from '../schema/index.js'
import { getOwnedSections } from './pack-loader.js'

// Items whose item-template produces this edge type from their parent
const STRUCTURAL_EDGE_BY_ITEM_TEMPLATE: Partial<Record<string, EdgeType>> = {
  Field: 'has-field',
  EnumValue: 'has-value',
}

type ClusterResult = {
  nodes: Map<string, Node>
  edgesByFrom: Map<string, Edge[]>
  edgesByTo: Map<string, Edge[]>
}

function addEdge(result: ClusterResult, edge: Edge): void {
  const from = result.edgesByFrom.get(edge.from) ?? []
  from.push(edge)
  result.edgesByFrom.set(edge.from, from)

  const to = result.edgesByTo.get(edge.to) ?? []
  to.push(edge)
  result.edgesByTo.set(edge.to, to)
}

function materialiseSection(
  parentId: string,
  sectionKey: string,
  sectionData: Record<string, unknown>,
  itemTemplateName: string,
  templates: Map<string, Template>,
  inheritedState: State,
  inheritedStability: Stability,
  schemaVersion: string,
  lastModifiedAt: string,
  component: string,
  result: ClusterResult,
  diagnostics: Diagnostic[],
  sourceFile: string,
): void {
  const itemTemplate = templates.get(itemTemplateName)
  if (!itemTemplate) {
    diagnostics.push({
      severity: 'error',
      file: sourceFile,
      nodeId: parentId,
      message: `unknown item-template: ${itemTemplateName} (section: ${sectionKey})`,
    })
    return
  }

  const childOwnedSections = getOwnedSections(itemTemplate)
  const childOwnedKeys = new Set(Object.keys(childOwnedSections))
  const structuralEdgeType = STRUCTURAL_EDGE_BY_ITEM_TEMPLATE[itemTemplateName] ?? null

  for (const [itemKey, itemRaw] of Object.entries(sectionData)) {
    const childId = `${parentId}.${sectionKey}.${itemKey}`
    const item = (itemRaw ?? {}) as Record<string, unknown>

    const childState: State = (item.state as State | undefined) ?? inheritedState
    const childStability: Stability = (item.stability as Stability | undefined) ?? inheritedStability

    // Extract properties: everything except owned section keys and state/stability
    const properties: Record<string, unknown> = {}
    for (const [k, v] of Object.entries(item)) {
      if (!childOwnedKeys.has(k) && k !== 'state' && k !== 'stability') {
        properties[k] = v
      }
    }

    const childNode: Node = {
      id: childId,
      template: itemTemplateName,
      component,
      state: childState,
      stability: childStability,
      schemaVersion,
      lastModifiedAt,
      properties,
    }
    result.nodes.set(childId, childNode)

    // Materialise structural edge from parent to this child
    if (structuralEdgeType !== null) {
      const edgeId = `${parentId}__${structuralEdgeType}__${childId}`
      addEdge(result, {
        id: edgeId,
        from: parentId,
        to: childId,
        type: structuralEdgeType,
        state: 'proposed',
        stability: 'unstable',
      })
    }

    // Recurse into this child's owned sections
    for (const [childSectionKey, childItemTemplateName] of Object.entries(childOwnedSections)) {
      const childSectionData = item[childSectionKey]
      if (childSectionData && typeof childSectionData === 'object') {
        materialiseSection(
          childId,
          childSectionKey,
          childSectionData as Record<string, unknown>,
          childItemTemplateName,
          templates,
          childState,
          childStability,
          schemaVersion,
          lastModifiedAt,
          component,
          result,
          diagnostics,
          sourceFile,
        )
      }
    }
  }
}

export async function loadClusters(
  graphPath: string,
  templates: Map<string, Template>,
  diagnostics: Diagnostic[],
): Promise<ClusterResult> {
  const result: ClusterResult = {
    nodes: new Map(),
    edgesByFrom: new Map(),
    edgesByTo: new Map(),
  }

  const componentsDir = path.join(graphPath, 'components')
  if (!existsSync(componentsDir)) return result

  for (const component of readdirSync(componentsDir)) {
    const componentDir = path.join(componentsDir, component)

    for (const templateFolder of readdirSync(componentDir)) {
      const folderPath = path.join(componentDir, templateFolder)

      const yamlFiles = readdirSync(folderPath).filter(f => f.endsWith('.yaml'))

      for (const yamlFile of yamlFiles) {
        const filePath = path.join(folderPath, yamlFile)
        let raw: unknown
        try {
          raw = parseYaml(readFileSync(filePath, 'utf-8'))
        } catch (err) {
          diagnostics.push({ severity: 'error', file: filePath, message: `failed to parse YAML: ${err}` })
          continue
        }

        const doc = raw as Record<string, unknown>

        // Validate required root node fields
        const requiredFields = ['id', 'template', 'component', 'state', 'stability', 'schemaVersion', 'lastModifiedAt'] as const
        const missing = requiredFields.filter(f => !(f in doc))
        if (missing.length > 0) {
          diagnostics.push({ severity: 'error', file: filePath, message: `missing required fields: ${missing.join(', ')}` })
          continue
        }

        const rootId = doc.id as string
        const templateName = doc.template as string
        const rootState = doc.state as State
        const rootStability = doc.stability as Stability
        const schemaVersion = doc.schemaVersion as string
        const lastModifiedAt = doc.lastModifiedAt as string

        const template = templates.get(templateName)
        if (!template) {
          diagnostics.push({ severity: 'error', file: filePath, nodeId: rootId, message: `unknown template: ${templateName}` })
          continue
        }

        const rootNode: Node = {
          id: rootId,
          template: templateName,
          component,
          state: rootState,
          stability: rootStability,
          schemaVersion,
          lastModifiedAt,
          properties: (doc.properties ?? {}) as Record<string, unknown>,
        }
        result.nodes.set(rootId, rootNode)

        // Materialise owned sections
        const ownedSections = getOwnedSections(template)
        for (const [sectionKey, itemTemplateName] of Object.entries(ownedSections)) {
          const sectionData = doc[sectionKey]
          if (sectionData && typeof sectionData === 'object') {
            materialiseSection(
              rootId,
              sectionKey,
              sectionData as Record<string, unknown>,
              itemTemplateName,
              templates,
              rootState,
              rootStability,
              schemaVersion,
              lastModifiedAt,
              component,
              result,
              diagnostics,
              filePath,
            )
          }
        }
      }
    }
  }

  return result
}
```

```typescript
import { readFileSync, readdirSync, existsSync } from 'node:fs'
import path from 'node:path'
import { parse as parseYaml } from 'yaml'
import type { Edge, EdgeType, State, Stability, Node, Diagnostic } from '../schema/index.js'

const VALID_EDGE_TYPES = new Set<string>([
  'triggers', 'produces', 'reads', 'calls', 'implements',
  'maps-to', 'derived-from', 'renamed-from', 'has-field', 'has-value',
])

type EdgeResult = {
  edgesByFrom: Map<string, Edge[]>
  edgesByTo: Map<string, Edge[]>
}

function addEdge(result: EdgeResult, edge: Edge): void {
  const from = result.edgesByFrom.get(edge.from) ?? []
  from.push(edge)
  result.edgesByFrom.set(edge.from, from)

  const to = result.edgesByTo.get(edge.to) ?? []
  to.push(edge)
  result.edgesByTo.set(edge.to, to)
}

export async function loadEdges(
  graphPath: string,
  nodes: Map<string, Node>,
  diagnostics: Diagnostic[],
): Promise<EdgeResult> {
  const result: EdgeResult = { edgesByFrom: new Map(), edgesByTo: new Map() }
  const edgesDir = path.join(graphPath, 'edges')
  if (!existsSync(edgesDir)) return result

  const files = readdirSync(edgesDir).filter(f => f.endsWith('.yaml'))

  for (const file of files) {
    const filePath = path.join(edgesDir, file)
    let raw: unknown
    try {
      raw = parseYaml(readFileSync(filePath, 'utf-8'))
    } catch (err) {
      diagnostics.push({ severity: 'error', file: filePath, message: `failed to parse YAML: ${err}` })
      continue
    }

    const doc = raw as { edges?: unknown[] }
    const edgeList = doc.edges ?? []

    for (const edgeRaw of edgeList) {
      const e = edgeRaw as Record<string, unknown>

      if (typeof e.from !== 'string' || typeof e.to !== 'string' || typeof e.type !== 'string') {
        diagnostics.push({ severity: 'error', file: filePath, message: 'edge missing required from, to, or type' })
        continue
      }

      if (!VALID_EDGE_TYPES.has(e.type)) {
        diagnostics.push({ severity: 'error', file: filePath, message: `invalid edge type: ${e.type}` })
        continue
      }

      if (!nodes.has(e.from)) {
        diagnostics.push({
          severity: 'error',
          file: filePath,
          message: `edge from unresolved node: ${e.from}`,
        })
      }
      if (!nodes.has(e.to)) {
        diagnostics.push({
          severity: 'error',
          file: filePath,
          message: `edge to unresolved node: ${e.to}`,
        })
      }

      const edge: Edge = {
        id: `${e.from}__${e.type}__${e.to}`,
        from: e.from,
        to: e.to,
        type: e.type as EdgeType,
        state: (e.state as State | undefined) ?? 'proposed',
        stability: (e.stability as Stability | undefined) ?? 'unstable',
        notes: e.notes as string | undefined,
      }

      addEdge(result, edge)
    }
  }

  return result
}
```

```typescript
import { readFileSync, existsSync } from 'node:fs'
import path from 'node:path'
import { parse as parseYaml } from 'yaml'
import type { Graph, LoadOptions, Diagnostic, Edge } from '../schema/index.js'
import { LoadError } from '../schema/index.js'
import { loadPacks } from './pack-loader.js'
import { loadClusters } from './cluster-loader.js'
import { loadEdges } from './edge-loader.js'

const DEFAULT_PACKS_PATH = '.corum/packs'

export async function loadGraph(options: LoadOptions): Promise<Graph> {
  const { graphPath, strict = true } = options
  const diagnostics: Diagnostic[] = []

  // Read graph.yaml to discover pack paths
  const graphYamlPath = path.join(graphPath, 'graph.yaml')
  let packDirs: string[] = []

  if (existsSync(graphYamlPath)) {
    const doc = parseYaml(readFileSync(graphYamlPath, 'utf-8')) as Record<string, unknown>
    const packs = (doc.templatePacks as Array<{ name: string; path: string }> | undefined) ?? []
    packDirs = packs.map(p => path.resolve(graphPath, p.path))
  } else {
    const packsRoot = options.packsPath ?? DEFAULT_PACKS_PATH
    diagnostics.push({
      severity: 'warning',
      file: graphYamlPath,
      message: `graph.yaml not found; using default packs path: ${packsRoot}`,
    })
  }

  // Phase 1: load template packs
  const templates = await loadPacks(packDirs, diagnostics)

  // Phase 2: load cluster files
  const clusterResult = await loadClusters(graphPath, templates, diagnostics)

  // Phase 3: load explicit edge files
  const edgeResult = await loadEdges(graphPath, clusterResult.nodes, diagnostics)

  // Merge structural edges (from clusters) + explicit edges (from edge files)
  function mergeEdgeMaps(
    target: Map<string, Edge[]>,
    source: Map<string, Edge[]>,
  ): void {
    for (const [key, edges] of source) {
      const existing = target.get(key) ?? []
      target.set(key, [...existing, ...edges])
    }
  }

  const edgesByFrom = new Map(clusterResult.edgesByFrom)
  const edgesByTo = new Map(clusterResult.edgesByTo)
  mergeEdgeMaps(edgesByFrom, edgeResult.edgesByFrom)
  mergeEdgeMaps(edgesByTo, edgeResult.edgesByTo)

  const graph: Graph = {
    nodesById: clusterResult.nodes,
    edgesByFrom,
    edgesByTo,
    templates,
    diagnostics,
  }

  const errors = diagnostics.filter(d => d.severity === 'error')
  if (strict && errors.length > 0) {
    throw new LoadError(diagnostics)
  }

  return graph
}
```

```typescript
import type { Graph, Node, Edge, State, Stability } from '../schema/index.js'
import { QueryError } from '../schema/index.js'

export type ListNodesFilter = {
  template?: string
  component?: string
  state?: State
  stability?: Stability
}

export type ClusterResult = {
  root: Node
  children: Node[]
  edges: Edge[]
}

export type LinkedFieldsResult = {
  edges: Edge[]
  nodes: Node[]
}

export function listNodes(graph: Graph, filter: ListNodesFilter = {}): Node[] {
  return [...graph.nodesById.values()].filter(node => {
    if (filter.template !== undefined && node.template !== filter.template) return false
    if (filter.component !== undefined && node.component !== filter.component) return false
    if (filter.state !== undefined && node.state !== filter.state) return false
    if (filter.stability !== undefined && node.stability !== filter.stability) return false
    return true
  })
}

export function getCluster(graph: Graph, nodeId: string): ClusterResult {
  const root = graph.nodesById.get(nodeId)
  if (!root) throw new QueryError(`Node not found: ${nodeId}`)

  const prefix = nodeId + '.'
  const children = [...graph.nodesById.values()].filter(n => n.id.startsWith(prefix))
  const clusterIds = new Set([nodeId, ...children.map(n => n.id)])

  const edges: Edge[] = []
  const seen = new Set<string>()
  for (const id of clusterIds) {
    for (const edge of graph.edgesByFrom.get(id) ?? []) {
      if (clusterIds.has(edge.to) && !seen.has(edge.id)) {
        edges.push(edge)
        seen.add(edge.id)
      }
    }
  }

  return { root, children, edges }
}

export function getLinkedFields(graph: Graph, nodeId: string): LinkedFieldsResult {
  const root = graph.nodesById.get(nodeId)
  if (!root) throw new QueryError(`Node not found: ${nodeId}`)

  const prefix = nodeId + '.'
  const ownedFieldIds = new Set(
    [...graph.nodesById.entries()]
      .filter(([id, node]) => id.startsWith(prefix) && node.template === 'Field')
      .map(([id]) => id),
  )

  const resultEdges: Edge[] = []
  const resultNodeIds = new Set<string>()
  const seen = new Set<string>()

  for (const fieldId of ownedFieldIds) {
    for (const edge of graph.edgesByFrom.get(fieldId) ?? []) {
      if (edge.type === 'maps-to' && !seen.has(edge.id)) {
        resultEdges.push(edge)
        resultNodeIds.add(edge.from)
        resultNodeIds.add(edge.to)
        seen.add(edge.id)
      }
    }
    for (const edge of graph.edgesByTo.get(fieldId) ?? []) {
      if (edge.type === 'maps-to' && !seen.has(edge.id)) {
        resultEdges.push(edge)
        resultNodeIds.add(edge.from)
        resultNodeIds.add(edge.to)
        seen.add(edge.id)
      }
    }
  }

  const nodes = [...resultNodeIds]
    .map(id => graph.nodesById.get(id))
    .filter((n): n is Node => n !== undefined)

  return { edges: resultEdges, nodes }
}
```

```typescript
import { Server } from '@modelcontextprotocol/sdk/server/index.js'
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js'
import { CallToolRequestSchema, ListToolsRequestSchema } from '@modelcontextprotocol/sdk/types.js'
import path from 'node:path'
import { loadGraph } from '../loader/index.js'
import { listNodes, getCluster, getLinkedFields, type ListNodesFilter } from '../graph/index.js'
import type { Graph } from '../schema/index.js'
import { QueryError } from '../schema/index.js'

type ToolResult = {
  content: Array<{ type: string; text: string }>
  isError?: boolean
}

export function createMcpHandlers(graph: Graph): {
  list_nodes: (args: Record<string, unknown>) => ToolResult
  get_cluster: (args: Record<string, unknown>) => ToolResult
  get_linked_fields: (args: Record<string, unknown>) => ToolResult
} {
  return {
    list_nodes(args) {
      const filter: ListNodesFilter = {
        template: args.template as string | undefined,
        component: args.component as string | undefined,
        state: args.state as ListNodesFilter['state'],
        stability: args.stability as ListNodesFilter['stability'],
      }
      const nodes = listNodes(graph, filter)
      const summaries = nodes.map(n => ({
        id: n.id,
        template: n.template,
        component: n.component,
        state: n.state,
        stability: n.stability,
      }))
      return { content: [{ type: 'text', text: JSON.stringify(summaries, null, 2) }] }
    },

    get_cluster(args) {
      try {
        const nodeId = args.node_id as string
        const cluster = getCluster(graph, nodeId)
        return { content: [{ type: 'text', text: JSON.stringify(cluster, null, 2) }] }
      } catch (err) {
        const message = err instanceof QueryError ? err.message : String(err)
        return { content: [{ type: 'text', text: message }], isError: true }
      }
    },

    get_linked_fields(args) {
      try {
        const nodeId = args.node_id as string
        const result = getLinkedFields(graph, nodeId)
        return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] }
      } catch (err) {
        const message = err instanceof QueryError ? err.message : String(err)
        return { content: [{ type: 'text', text: message }], isError: true }
      }
    },
  }
}

// Entry point — only runs when executed directly
if (process.argv[1] && import.meta.url.endsWith(process.argv[1].replace(/\\/g, '/'))) {
  const graphPath = process.env.CORUM_GRAPH_PATH ?? path.join(process.cwd(), '.corum/graph')

  let graph: Graph
  let loadError: string | undefined

  try {
    graph = await loadGraph({ graphPath, strict: true })
  } catch (err) {
    loadError = String(err)
    graph = await loadGraph({ graphPath, strict: false }).catch(() => ({
      nodesById: new Map(),
      edgesByFrom: new Map(),
      edgesByTo: new Map(),
      templates: new Map(),
      diagnostics: [],
    }))
  }

  const handlers = createMcpHandlers(graph)

  const server = new Server(
    { name: 'corum', version: '0.1.0' },
    { capabilities: { tools: {} } },
  )

  server.setRequestHandler(ListToolsRequestSchema, async () => ({
    tools: [
      {
        name: 'list_nodes',
        description: 'List nodes in the graph. Returns id, template, component, state, stability for each matched node.',
        inputSchema: {
          type: 'object',
          properties: {
            template: { type: 'string', description: 'Filter by template name (e.g. DomainModel, APIEndpoint)' },
            component: { type: 'string', description: 'Filter by component (bounded context) name' },
            state: { type: 'string', description: 'Filter by state (draft, proposed, agreed, future, removed, implemented)' },
            stability: { type: 'string', description: 'Filter by stability (unstable, stable, deprecated)' },
          },
        },
      },
      {
        name: 'get_cluster',
        description: 'Get the full cluster for a root node: the node itself, all owned child nodes, and edges between them.',
        inputSchema: {
          type: 'object',
          required: ['node_id'],
          properties: {
            node_id: { type: 'string', description: 'Fully qualified node ID (e.g. orders.DomainModel.order)' },
          },
        },
      },
      {
        name: 'get_linked_fields',
        description: 'Get all maps-to edges touching fields owned by the given root node, with both endpoint nodes.',
        inputSchema: {
          type: 'object',
          required: ['node_id'],
          properties: {
            node_id: { type: 'string', description: 'Fully qualified root node ID' },
          },
        },
      },
    ],
  }))

  server.setRequestHandler(CallToolRequestSchema, async request => {
    if (loadError) {
      return {
        content: [{ type: 'text', text: `Graph load error: ${loadError}` }],
        isError: true,
      }
    }

    const { name, arguments: args } = request.params
    const toolArgs = (args ?? {}) as Record<string, unknown>

    switch (name) {
      case 'list_nodes':
        return handlers.list_nodes(toolArgs)
      case 'get_cluster':
        return handlers.get_cluster(toolArgs)
      case 'get_linked_fields':
        return handlers.get_linked_fields(toolArgs)
      default:
        return {
          content: [{ type: 'text', text: `Unknown tool: ${name}` }],
          isError: true,
        }
    }
  })

  const transport = new StdioServerTransport()
  await server.connect(transport)
}
```
